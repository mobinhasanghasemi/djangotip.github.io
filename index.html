Lazy Evaluation در QuerySet

جنگو تا زمانی که واقعاً به داده نیاز نداشته باشی، کوئری نمی‌زنه. این اولین و مهم‌ترین اصل بهینه‌سازیه.

⚡ درک مفهوم Lazy بودن

QuerySet‌های جنگو تنبل هستن — تا زمان evaluation کوئری به دیتابیس ارسال نمیشه.

✗ کوئری اضافی — بدون استفاده
# این خط هیچ کوئری‌ای به DB نمی‌زنه!
qs = User.objects.filter(is_active=True)

# ولی این خط کوئری می‌زنه چون evaluation شده
list(qs)  # SELECT * FROM user WHERE is_active

# ❌ اشتباه رایج: تبدیل به list قبل از نیاز
users = list(User.objects.all())
for u in users:
    if u.age > 18:
        print(u.name)
🐌 بارگذاری کل جدول در حافظه
✓ فیلتر در سطح دیتابیس
# ✓ بهینه: فیلتر در DB انجام میشه
users = User.objects.filter(
    is_active=True,
    age__gt=18
)

# فقط وقتی iteration بشه کوئری اجرا میشه
for u in users:
    print(u.name)

# SELECT * FROM user
# WHERE is_active=1 AND age > 18
⚡ فیلتر در سمت DB
نکته کلیدی: هیچوقت QuerySet رو بدون نیاز به list() تبدیل نکن. بذار جنگو خودش تصمیم بگیره کی کوئری بزنه. chaining فیلترها هم هزینه‌ای نداره چون هنوز evaluation نشده.
⚠️ تله‌های رایج QuerySet
✗ ارزیابی چندباره
# ❌ هر بار یه کوئری جدید!
if User.objects.filter(active=True).exists():
    users = User.objects.filter(active=True)
    for u in users:
        process(u)

# ۳ کوئری جداگانه اجرا میشه!
✓ کش کردن خودکار QuerySet
# ✓ کوئری فقط یکبار اجرا میشه
users = User.objects.filter(active=True)

if users:
    for u in users:
        process(u)

# یا explicit cache:
users = list(User.objects.filter(...))

select_related — حل N+1

معروف‌ترین مشکل性能ی جنگو: N+1 Query Problem. با select_related با یه JOIN حلش کن.

🔥 مشکل N+1 — بدترین دشمن性能
✗ مشکل N+1 — ۱۰۱ کوئری!
# مدل‌ها
class Author(models.Model):
    name = models.CharField(max_length=100)

class Book(models.Model):
    title = models.CharField(max_length=200)
    author = models.ForeignKey(Author, ...)

# ❌ ویو — N+1 problem
books = Book.objects.all()[:100]
for book in books:
    print(book.author.name)
    # هر iteration = 1 کوئری اضافه!
    # 1 + 100 = 101 کوئری
🐌 101 کوئری به دیتابیس
✓ select_related — فقط ۱ کوئری
# ✓ با JOIN — فقط ۱ کوئری
books = Book.objects.select_related(
    'author'
).all()[:100]

for book in books:
    print(book.author.name)
    # بدون کوئری اضافه!

# SQL:
# SELECT book.*, author.*
# FROM book
# INNER JOIN author
#   ON book.author_id = author.id
# LIMIT 100
⚡ 1 کوئری با JOIN
101
کوئری بدون بهینه‌سازی
1
کوئری با select_related
99%
کاهش تعداد کوئری
💡 کی استفاده کنیم: select_related برای ForeignKey و OneToOneField کار می‌کنه. از LEFT OUTER JOIN استفاده می‌کنه. برای روابط ManyToMany کار نمی‌کنه — اونجا باید از prefetch_related استفاده کنی.
🔗 زنجیره‌ای کردن select_related 
# روابط عمیق رو هم میشه زنجیره کرد
order = Order.objects.select_related(
    'customer',              # FK
    'customer__address',     # FK → FK
    'product__category',     # FK → FK
).get(id=1)

# یک کوئری با چند JOIN
# SELECT order.*, customer.*, address.*,
#        product.*, category.*
# FROM order
# INNER JOIN customer ON ...
# LEFT JOIN address ON ...
# INNER JOIN product ON ...
# LEFT JOIN category ON ...

prefetch_related — روابط Many

برای ManyToMany و reverse ForeignKey. دو کوئری جدا ولی بهینه.

🔄 prefetch_related برای M2M
✗ بدون prefetch — N+1
class Article(models.Model):
    tags = models.ManyToManyField('Tag')
    title = models.CharField()

# ❌ هر مقاله یه کوئری برای تگ‌ها
articles = Article.objects.all()
for a in articles:
    print(a.tags.all())
    # N+1 problem
✓ با prefetch — فقط ۲ کوئری
# ✓ فقط ۲ کوئری اجرا میشه
articles = Article.objects.prefetch_related(
    'tags'
).all()

for a in articles:
    print(a.tags.all())
    # از کش داخلی استفاده میشه

# کوئری ۱: SELECT * FROM article
# کوئری ۲: SELECT * FROM tag
#          INNER JOIN article_tags
#          WHERE article_id IN (...)
🎯 Prefetch پیشرفته با Prefetch Object 
from django.db.models import Prefetch

# فیلتر کردن پری‌فچ — فقط تگ‌های فعال
articles = Article.objects.prefetch_related(
    Prefetch(
        'tags',
        queryset=Tag.objects.filter(is_active=True),
        to_attr='active_tags'  # ذخیره در attribute سفارشی
    )
)

for a in articles:
    print(a.active_tags)  # لیست آماده، بدون کوئری اضافه

# ترکیب select_related + prefetch_related
orders = Order.objects.select_related(
    'customer'          # FK → JOIN
).prefetch_related(
    'items',             # M2M → separate query
    'items__product'     # M2M → FK → separate query
)
💡 select_related vs prefetch_related: select_related = JOIN (یک کوئری) برای FK/OneToOne. prefetch_related = کوئری جدا (دو کوئری) برای M2M/reverse FK. در روابط عمیق ترکیب هر دو ایده‌آله.

only و defer — بارگذاری انتخابی

وقتی فقط چند فیلد لازم داری، کل ردیف رو بارگذاری نکن.

📊 only() — فقط فیلدهای مورد نیاز
✗ بارگذاری همه فیلدها
# ❌ SELECT * — تمام فیلدها
users = User.objects.all()

for u in users:
    print(f"{u.first_name} {u.last_name}")
    # email, avatar, bio, ...
    # همه بارگذاری شدن ولی استفاده نشدن
✓ فقط فیلدهای لازم
# ✓ SELECT id, first_name, last_name
users = User.objects.only(
    'first_name',
    'last_name'
)

for u in users:
    print(f"{u.first_name} {u.last_name}")
🧊 defer() — حذف فیلدهای سنگین 
# defer = "این فیلدها رو بعداً بارگذاری کن"
articles = Article.objects.defer(
    'content',      # متن کامل مقاله — سنگین
    'raw_html'      # HTML خام — سنگین
).all()

for a in articles:
    print(a.title)     # ✓ بدون کوئری اضافه
    print(a.summary)   # ✓ بدون کوئری اضافه
    print(a.content)   # ⚠️ کوئری اضافه — lazy load

# ترکیب با select_related
posts = Post.objects.select_related(
    'author'
).only(
    'title',
    'created_at',
    'author__first_name'
)
⚠️ احتیاط: اگه به فیلد deferred دسترسی پیدا کنی، جنگو یه کوئری اضافه می‌زنه. فقط وقتی از defer/only استفاده کن که مطمئنی به اون فیلدها نیاز نداری.

annotate و aggregate — محاسبه در DB

به جای حلقه در پایتون، محاسبات رو به دیتابیس بسپار.

🧮 annotate — اضافه کردن فیلد محاسباتی
✗ محاسبه در Python — N+1 و کند
# ❌ شمارش سفارشات در پایتون
authors = Author.objects.all()
result = []
for a in authors:
    count = a.books.count()  # 1 کوئری
    result.append({
        'name': a.name,
        'book_count': count
    })

# N+1 کوئری + حلقه در Python
🐌 N+1 + CPU Python
✓ annotate در DB — ۱ کوئری
from django.db.models import Count

# ✓ همه چیز در یک کوئری
authors = Author.objects.annotate(
    book_count=Count('books')
).values('name', 'book_count')

# SQL:
# SELECT a.name, COUNT(b.id) as book_count
# FROM author a
# LEFT JOIN book b ON b.author_id = a.id
# GROUP BY a.id, a.name
⚡ 1 کوئری بهینه
📈 توابع مفید annotate 
from django.db.models import (
    Count, Sum, Avg, Max, Min, F, Q, Value,
    Case, When, CharField
)
from django.db.models.functions import (
    Concat, Lower, Upper, Coalesce, Now
)

# مثال‌های کاربردی
products = Product.objects.annotate(
    # شمارش
    review_count=Count('reviews'),
    # میانگین
    avg_rating=Avg('reviews__rating'),
    # محاسبه با F expressions
    profit=F('price') - F('cost'),
    # شرطی
    status=Case(
        When(stock__gt=0, then=Value('موجود')),
        default=Value('ناموجود'),
        output_field=CharField()
    ),
    # کانکت
    full_name=Concat('first_name', Value(' '), 'last_name'),
).filter(
    avg_rating__gte=4.0
)

Index‌ها — سرعت جستجو را ۱۰۰x کن

بدون ایندکس، دیتابیس باید کل جدول رو اسکن کنه. با ایندکس مناسب، جستجو O(log n) میشه.

📇 انواع Index در جنگو 
class Product(models.Model):
    name = models.CharField(max_length=200)
    slug = models.SlugField(unique=True)  # خودش ایندکس داره
    price = models.DecimalField(max_digits=10, decimal_places=2)
    category = models.ForeignKey(Category, ...)  # خودش ایندکس داره
    created_at = models.DateTimeField(auto_now_add=True)
    is_active = models.BooleanField(default=True)
    brand = models.CharField(max_length=100)

    class Meta:
        # ① تک‌فیلدی
        indexes = [
            models.Index(fields=['price']),
            models.Index(fields=['created_at']),

            # ② ترکیبی — برای کوئری‌های چندفیلدی
            models.Index(
                fields=['category', 'price'],
                name='idx_cat_price'
            ),

            # ③ شرطی — فقط ردیف‌های مرتبط ایندکس میشن
            models.Index(
                condition=Q(is_active=True),
                fields=['name'],
                name='idx_active_name'
            ),

            # ④ Partial Index با PostgreSQL
            models.Index(
                fields=['brand'],
                condition=Q(price__gt=100),
                name='idx_expensive_brand'
            ),
        ]
💡 قانون طلایی: فقط فیلدهایی که در .filter()، .order_by() و .exclude() استفاده میشن رو ایندکس کن. ایندکس زیاد = کند شدن INSERT/UPDATE.
🔍 Gin Index برای جستجوی متنی (PostgreSQL) 
from django.contrib.postgres.indexes import GinIndex
from django.contrib.postgres.search import SearchVectorField

class Article(models.Model):
    title = models.CharField(max_length=300)
    content = models.TextField()
    search_vector = models.SearchVectorField(null=True)

    class Meta:
        indexes = [
            GinIndex(
                fields=['search_vector'],
                name='idx_search_vector'
            )
        ]

# جستجوی تمام‌متنی سریع
articles = Article.objects.filter(
    search_vector=SearchQuery('جنگو بهینه‌سازی')
)

Raw SQL — وقتی ORM کافی نیست

گاهی برای کوئری‌های پیچیده، RAW SQL بهترین گزینه‌ست.

🛠️ روش‌های مختلف Raw SQL 
# ① raw() — ساده‌ترین روش
users = User.objects.raw('''
    SELECT id, first_name, last_name,
           COUNT(order.id) as order_count
    FROM auth_user u
    LEFT JOIN shop_order o ON o.user_id = u.id
    GROUP BY u.id
    HAVING COUNT(o.id) > 5
    ORDER BY order_count DESC
''')

# ② connection.cursor — برای کوئری‌های غیرSELECT
from django.db import connection

with connection.cursor() as cursor:
    cursor.execute("""
        WITH monthly_sales AS (
            SELECT
                date_trunc('month', created_at) AS month,
                SUM(total) AS revenue
            FROM shop_order
            WHERE created_at >= NOW() - INTERVAL '1 year'
            GROUP BY 1
        )
        SELECT month, revenue,
               LAG(revenue) OVER (ORDER BY month) AS prev_month,
               revenue - LAG(revenue) OVER (ORDER BY month) AS growth
        FROM monthly_sales
        ORDER BY month DESC
    """)
    results = dictfetchall(cursor)

# ③ تابع کمکی
def dictfetchall(cursor):
    columns = [col[0] for col in cursor.description]
    return [
        dict(zip(columns, row))
        for row in cursor.fetchall()
    ]
⚠️ SQL Injection: هیچوقت رشته‌ها رو مستقیم در SQL قرار نده. از پارامتر استفاده کن: cursor.execute("... WHERE id = %s", [user_id])

from django.core.cache import caches

def product_list(request):
    cache = caches['default']
    cache_key = 'product_list_page_' + request.GET.get('page', '1')

    data = cache.get(cache_key)
    if data is None:
        data = list(
            Product.objects.filter(is_active=True)
            .select_related('category')
            .values('id', 'name', 'price', 'category__name')
        )
        cache.set(cache_key, data, timeout=60)

    return JsonResponse(data, safe=False)

بهینه‌سازی پایگاه داده — PostgreSQL & SQLite

تنظیمات سطح دیتابیس که تأثیر مستقیم روی عملکرد دارن. WAL Mode، Shared Buffers، و خیلی بیشتر.

🗄️ SQLite — بهینه‌سازی حرفه‌ای
✗ پیش‌فرض SQLite — کند
# settings.py — پیش‌فرض
DATABASES = {
    'default': {
        'ENGINE': 'django.db.backends.sqlite3',
        'NAME': BASE_DIR / 'db.sqlite3',
    }
}

# مشکلات:
# - WAL غیرفعال = قفل نوشتن
# - Journal Mode = DELETE (کند)
# - بدون busy_timeout
# - Synchronous = FULL (کند)
✓ SQLite بهینه‌شده
# settings.py — بهینه
DATABASES = {
    'default': {
        'ENGINE': 'django.db.backends.sqlite3',
        'NAME': BASE_DIR / 'db.sqlite3',
        'OPTIONS': {
            'transaction_mode': 'IMMEDIATE',
        },
    }
}

# در AppConfig.ready() یا startup:
import sqlite3
conn = sqlite3.connect(db_path)
conn.execute('PRAGMA journal_mode=WAL')
conn.execute('PRAGMA synchronous=NORMAL')
conn.execute('PRAGMA cache_size=-64000')  # 64MB
conn.execute('PRAGMA busy_timeout=5000')
conn.execute('PRAGMA temp_store=MEMORY')
conn.execute('PRAGMA mmap_size=268435456')  # 256MB
PRAGMA‌های حیاتی SQLite:
• journal_mode=WAL — Write-Ahead Logging: خواندن و نوشتن همزمان ممکن میشه. سرعت نوشتن ۲-۵x بهتر.
• synchronous=NORMAL — به جای FULL، فقط در نقاط خاص sync می‌کنه. سرعت ۲x بهتر با امنیت کافی.
• cache_size=-64000 — ۶۴ مگابایت کش. منفی یعنی کیلوبایت.
• busy_timeout=5000 — ۵ ثانیه صبر قبل از خطا. از SQLITE_BUSY جلوگیری می‌کنه.
• mmap_size=256MB — Memory-mapped I/O: خواندن فایل دیتابیس مستقیم از حافظه.
📝 WAL Mode عمیق‌تر — چرا و چگونه 
# ── درک WAL Mode ─────────────────────────────
#
# DELETE Mode (پیش‌فرض):
#   نوشتن → قفل کل دیتابیس → خواندن منتظر می‌مونه
#   ┌──────┐

class SqliteConfig:
    """اعمال تنظیمات SQLite هر بار اتصال"""

    @staticmethod
    def configure(connection):
        cursor = connection.cursor()
        cursor.execute('PRAGMA journal_mode=WAL')
        cursor.execute('PRAGMA synchronous=NORMAL')
        cursor.execute('PRAGMA cache_size=-64000')
        cursor.execute('PRAGMA busy_timeout=5000')
        cursor.execute('PRAGMA temp_store=MEMORY')
        cursor.execute('PRAGMA mmap_size=268435456')
        cursor.execute('PRAGMA wal_autocheckpoint=1000')

تنظیمات مهم

  • shared_buffers: ۲۵٪ RAM سیستم. کش اصلی PostgreSQL.
  • work_mem: حافظه برای sort/hash operations. مراقب باش زیاد نباشه چون per-connection هست.
  • effective_cache_size: ۷۵٪ RAM. به planner کمک می‌کنه تصمیم بگیره از ایندکس استفاده کنه یا seq scan.
  • random_page_cost: برای SSD = 1.1 (پیش‌فرض 4.0 برای HDD).
  • WAL: Write-Ahead Log برای durability و replication.
  • Autovacuum: فضای مرده رو پاک می‌کنه. بدونش DB کند میشه.

# ── Django Debug Toolbar + EXPLAIN ──────────
from django.db import connection

def explain_query(queryset):
    """EXPLAIN ANALYZE روی هر QuerySet"""
    sql, params = queryset.query.sql_with_params()
    with connection.cursor() as cursor:
        cursor.execute(
            f"EXPLAIN ANALYZE {sql}",
            params
        )
        return cursor.fetchall()

# استفاده:
qs = Product.objects.filter(category_id=5).order_by('-created_at')
for line in explain_query(qs):
    print(line)

# خروجی نمونه:
# Seq Scan on product  (cost=0.00..1234.56 rows=100)
#   Filter: (category_id = 5)
#   Rows Removed by Filter: 9900
#
# ⚠️ اگه Seq Scan دیدی و جدول بزرگه → ایندکس لازمه!
💡 نکات EXPLAIN: Seq Scan = اسکن کل جدول (بد برای جدول بزرگ). Index Scan = استفاده از ایندکس (خوب). Nested Loop = JOIN. Hash Join = JOIN بزرگ. همیشه EXPLAIN ANALYZE بزن تا زمان واقعی رو ببینی.

@receiver(post_save, sender=Order)
def update_inventory(sender, instance, **kwargs):
    if not instance._state.adding:
        return  # فقط برای create

    items = instance.items.select_related(
        'product'
    )
    products = []
    for item in items:
        item.product.stock = F('stock') - item.quantity
        products.append(item.product)

class OptimizedMiddleware:
    def __init__(self, get_response):
        self.get_response = get_response

    def __call__(self, request):
        # ✓ فقط اگه user لاگین کرده
        if request.user.is_authenticated:

        logger.debug(
            "%s %s %d",
            request.method,
            request.path,
            response.status_code
        )

        return response

# ── ④ Custom Query Logger ──────────────────
import time
import logging

db_logger = logging.getLogger('django.db.backends')

class QueryCountMiddleware:
    def __init__(self, get_response):
        self.get_response = get_response

    def __call__(self, request):
        from django.db import connection
        connection.queries_log.clear()

        start = time.time()
        response = self.get_response(request)
        duration = time.time() - start

        query_count = len(connection.queries)
        if query_count > 20:
            db_logger.warning(
                f"⚠️ {request.path}: {query_count} queries, "
                f"{duration:.2f}s"
            )

        return response
